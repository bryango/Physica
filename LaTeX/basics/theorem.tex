	,spaceabove=.8\baselineskip
	,spacebelow=.8\baselineskip
	,notefont=\bfseries

	,postheadspace=.6em

		,leftmargin=10

		,innerbottommargin=\baselineskip

]{theorem*}

	,style=nonumber

]{definition*}
